PRIMERA PARTE DE GIT
git config user.name					::::: ver el nombre del usuario

git config user.email					::::: ver el mail del usuario
git config --global user.email "zenid77@gmail.com"	::::: configurar el mail del usuario
pwd							::::: imprime la ruta del directorio actual
ls							::::: lista del contenido del directorio actual
cd							::::: cambiar de directorio
cd ..							::::: retroceder en la ruta
mkdir							::::: crear directorio (make directory)
git init						::::: convertir directorio en repositorio git
git init dir						::::: convertir directorio en repositorio git dentro del directorio donde estabamos
git status						::::: ver como esta el area de trabajo
git add							::::: añadir al staging area todos los archivos del directorio
git add file						::::: añadir al staging area un archivos del directorio
git rm --cached file					::::: remover del staging area y devoilverlo al working directory un archivo
git commit -m "label"					::::: enviar los archivos del staging area al repositorio local con etiqueta
git commit -am "label"					::::: enviar archivos del working directory directos al repositorio local
git log							::::: visualizar el historial de commits con la identificación, autor, fecha y mensaje/etiqueta
git push						::::: subir los archivos del repositorio local al repositorio remoto
git remote -v						::::: ver cuantos repositorios remotos estan asociados al repositorio local
git remote add origin URL				::::: asociar un repositorio remoto con el repositorio local
git branch -M main					::::: ---tiene que ver con la rama principal
git push -u origin main					::::: ---hace que iniciemos sesión en el nuevo repositorio
git pull						::::: cargar los archivos del repositorio remoto al local
git pull origin main					::::: cargar los commits de la rama main
git clone URL						::::: clonar un repositorio remoto entero

SEGUNDA PARTE DE GIT
git log --graph --oneline --decorate			::::: visualizar el historial de commits en lineas solo con las etiquetas/comentarios
git checkout code-commit				::::: restaurar el git cuando code-commit hacia adelante y hacia atrás temporalmente
git log --graph --oneline --decorate --reflog		::::: visualizar el historial con los futuros commits si se viene de hacer un checkout al pasado
git checkout main					::::: restaurar el git al estado de la rama principal
git show code-commit					::::: visualizar los cambios realizados en el codigo de los archivos en el code-commit
git branch						::::: Ver las ramas temporales
git branch new-branch					::::: crear una rama temporal
git checkout the-branch					::::: situarse sobre la rama temporal
git push --set-upstream origin the-branch		::::: subimos la rama al repositorio remoto
git log --all --oneline --decorate --graph		::::: ver todos los commit del repositorio local
git merge the-branch					::::: fusionar las ramas sobreescribiendo el historial con el de the-branch ajeno
git merge --no-ff branch1 -m 'comentario'		::::: fusionar las ramas pero dejando un commit diferenciadolas en el historial
